import sys


def calcSum(ivNum1, ivNum2):
    suma = 0
    for x in range(int(ivNum1), int(ivNum2)+1):
        x = str(x)
        for y in x:
            suma += int(y)
    return suma


# Main function
def main():
    for tempInput in sys.stdin:
        # Creating variable
        tempArray = tempInput.split()
        lvNum1 = tempArray[0]
        lvNum2 = tempArray[1]
        if lvNum1 == "-1" and lvNum2 == "-1":
            return 0
        print(calcSum(lvNum1, lvNum2))


# Start of program
if __name__ == '__main__':
    main()
